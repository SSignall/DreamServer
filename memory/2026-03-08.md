**Key fix pattern applied:**
```javascript
function timingSafeEqual(a, b) {
  if (typeof a !== 'string' || typeof b !== 'string') return false;
  const MAX_LEN = 256;
  if (a.length > MAX_LEN || b.length > MAX_LEN) return false;
  
  let result = 0;
  // Constant-time comparison: always iterate MAX_LEN, use 0 for out-of-bounds
  for (let i = 0; i < MAX_LEN; i++) {
    const charA = i < a.length ? a.charCodeAt(i) : 0;
    const charB = i < b.length ? b.charCodeAt(i) : 0;
    result |= charA ^ charB;
  }
  // Include length check in constant-time result
  result |= (a.length === b.length) ? 0 : 1;
  return result === 0;
}
```

**Previously buggy patterns that were fixed:**
1. Using `padEnd(MAX_LEN, ' ')` or `padEnd(MAX_LEN, '\u0000')` - allowed prefix matches
2. Using `a + '\u0000'.repeat(MAX_LEN - a.length)` - same issue
3. Using `a.length ^ b.length` XOR instead of explicit `result |= (a.length === b.length) ? 0 : 1`
